// =============================================================
// macrodb — Schema V8 PROPOSAL (category-tree collapse + source groups)
//
// STATUS: PROPOSAL ONLY. db_er.txt (V7) remains the source of truth until an
// ADR accepts this. This file shows the DELTA from V7, not the whole schema.
//
// Drives three changes discussed 2026-06-16:
//   A. Collapse concepts + indicators + indicator_variants + tags + concept_tags
//      into ONE category TREE (strict single-parent). concept = attachable leaf;
//      indicator = derived (category × geography) query; variant = series
//      methodology + is_default; topic = ancestor walk. Series attaches to its
//      most-specific category node (single FK, NOT M:N topical tagging).
//   B. Keep series_hierarchy_edges (canonical decomposition); ADD provider-grain
//      source_groups + source_group_members (typed, self-nesting publication
//      layout: releases, tables, datasets, dashboards).
//   C. Graph DB — deferred, not in this proposal.
//
// Decisions locked 2026-06-16/17:
//   - Categories are a STRICT TREE, but the parent/child relation is kept in its
//     own adjacency table (category_edges), not a self-FK column — uniform with
//     the other membership tables and edges can carry sort_order. Tree semantics
//     are preserved by UNIQUE(child_category_id): at most one parent per node.
//     Drop that one constraint the day a DAG is wanted. Cost: parent/ancestor
//     lookups are a join, not a column read (marginal).
//   - Provider grouping is GENERIC source_groups (typed + self-nesting), not a
//     narrow statistical_tables. Handles FRED release→table nesting and lets a
//     series_source belong to many groups.
//
// Supersedes ADR 0022 (concept-grained tags) in full.
// Reverses ADR 0020 indicator-grain embedding; concept embedding moves to the
//   category node. Series embedding unchanged.
// Reverses ADR 0021 (series-family → indicator rename): indicator ceases to be
//   a stored entity.
// Extends ADR 0010 (canonical series hierarchy stays; groups added beside it).
//
// Migration cost: LOW. seed/data/ seeds only geographies, memberships,
//   providers, tags. concepts/indicators/variants are GENERATED by
//   bootstrap/fred_us_macro.py + services/registration.py, so the catalog is
//   regenerable: drop-and-rebootstrap, same workflow ADR 0022 used.
// =============================================================


// =======================
// NEW — replaces concepts, tags, concept_tags, indicators, indicator_variants
// =======================

categories [icon: folder-tree, color: purple] {
  id uuid pk
  code string unique              // canonical UPPERCASE SCREAMING_SNAKE; PRICES, CONSUMER_PRICES, CPI_ALL_ITEMS
  name string                     // human display label
  description string              // nullable
  kind string                     // ENUM: topic | concept
                                  //   topic   = browse/navigation node, NOT attachable by series (PRICES, CONSUMER_PRICES)
                                  //   concept = attachable economic idea, the old `concept` grain (CPI_ALL_ITEMS)
                                  //   Depth ≤ 3 is a CURATION CONVENTION (like the code-casing rule), not a CHECK.
                                  //   A concept node is normally a LEAF; a topic node groups children. (Convention,
                                  //   not enforced: nothing structurally forbids a concept node having children.)
  embedding vector                // nullable; 1536-dim; semantic search; populated for kind=concept; HNSW cosine ix_categories_embedding_hnsw
  embedding_model string          // nullable
  embedding_input_hash string     // nullable
  created_at timestamp
  updated_at timestamp
  // kind is informational + app-validated. Series-attach-to-concept-only is
  // enforced in Pydantic/service code, not by a DB constraint (downgraded 2026-06-17).
}

// Parent/child relation as a separate adjacency table (not a self-FK column).
// STRICT TREE today, enforced by UNIQUE(child_category_id) = at most one parent
// per node. Roots simply have no edge row. Drop that UNIQUE to allow a DAG later.
// A series' topics = the chain of ancestors walked up these edges.
category_edges [icon: git-branch, color: purple] {
  id uuid pk
  parent_category_id uuid fk      // FK -> categories.id; parent_category_id != child_category_id
  child_category_id uuid fk       // FK -> categories.id; UNIQUE(child_category_id) -> single parent = tree
  sort_order int                  // nullable; stable sibling ordering for browse
  created_at timestamp
  updated_at timestamp
}


// =======================
// CHANGED — series gains category attachment + default flag
// =======================
//
// series KEEPS every methodology column from V7 (geography_id, frequency,
// unit_*, measure, seasonal_adjustment, reference_*, embedding, etc.).
// Only the category linkage + is_default are new. Shown here are the deltas:

series (DELTA) {
  // ... all V7 columns unchanged ...
  category_id uuid fk             // FK -> categories.id; the MOST-SPECIFIC node this series belongs to.
                                  //   Encouraged ("lowest matchable level"), nullable to allow draft series
                                  //   pre-classification AND a liberal "not classified" state.
                                  //   RULE: must reference a kind=concept node (not a topic bucket).
                                  //   ENFORCED AT THE APP LAYER (Pydantic + service validation), NOT the DB.
                                  //   The composite-FK guardrail was considered and deliberately downgraded:
                                  //   tightening to a DB constraint later is a cheap backfill-and-validate pass.
  is_default boolean              // default reading within (category_id, geography_id) — the old
                                  //   indicator_variants.is_default. NO partial-unique enforced, consistent
                                  //   with V7's deliberate choice not to unique-constrain default variants.
}
// REMOVED FROM THE GRAPH: series no longer reaches topic via
//   indicator_variants -> indicators -> concepts -> concept_tags.
//   Topic now = ancestors(series.category_id) walked up category_edges.


// =======================
// NEW (Part B) — source groups: provider-side publication layout, generic
// =======================
//
// A source_group is a PUBLICATION/BROWSE unit as the provider presents it:
// a FRED release, a release TABLE, a SingStat dataset, a dashboard. Typed and
// self-nesting (release CONTAINS tables). Deliberately distinct from:
//   - series_hierarchy_edges  (canonical, provider-NEUTRAL analytical decomposition)
//   - ingestion_feed          (runtime FETCH unit; often coincides but not always)
// Keep all three separate or they rot into each other. Structure is NOT stored
// here — only membership + publication ordering. Indentation is derived by
// intersecting membership with series_hierarchy_edges (no parent_member_id).

source_groups [icon: folder, color: blue] {
  id uuid pk
  provider_catalog_id uuid fk     // FK -> provider_catalogs.id
  parent_group_id uuid            // nullable; FK -> source_groups.id; self-nesting (release -> table). parent != id
  group_type string               // ENUM: release | table | dataset | dashboard | other
  code string                     // nullable; provider's id e.g. "rid=21"; UNIQUE (provider_catalog_id, code)
  name string                     // e.g. "Employment Situation", "Table B-1"
  source_url string               // nullable; the published page
  notes string                    // nullable
  created_at timestamp
  updated_at timestamp
}

source_group_members [icon: rows, color: blue] {
  id uuid pk
  source_group_id uuid fk         // FK -> source_groups.id
  series_source_id uuid fk        // FK -> series_sources.id; the provider representation appearing in this group.
                                  //   UNIQUE (source_group_id, series_source_id)
  row_label string                // nullable; label as printed (may differ from series.name)
  sort_order int                  // nullable; order within the group
  created_at timestamp
  updated_at timestamp
}
// A series_source MAY belong to many groups (M:N), so it is NOT fixed to one
// table — it can appear in a release, a table within it, and a dashboard at once.

// DEFERRED (additive, NOT built now) — series_collections:
//   The canonical-side mirror of source_groups. A user-curated, cross-provider
//   grouping whose members are canonical `series` (NOT series_sources), so the
//   same logical series is never duplicated per provider. Kept SEPARATE from
//   source_groups on purpose: the two sit on opposite sides of macrodb's
//   canonical-vs-provider seam, and unifying them would force either a
//   polymorphic member FK (loses integrity) or collapse provider fidelity.
//   Shared structure (name, parent nesting, member sort_order) can be a shared
//   ORM mixin without sharing a table. Build when a real curation need appears.

// =============================================================
// WORKED EXAMPLE — the old US_CPI indicator, in the collapsed model
// =============================================================
//
// categories:
//   (PRICES,           topic,   emb=null)     <- root, no edge
//   (CONSUMER_PRICES,  topic,   emb=null)
//   (CPI_ALL_ITEMS,    concept, emb=<vec>)     <- attachable; old `concept`
//   (CPI_CORE,         concept, emb=<vec>)
// category_edges (child -> parent):
//   CONSUMER_PRICES -> PRICES
//   CPI_ALL_ITEMS   -> CONSUMER_PRICES
//   CPI_CORE        -> CONSUMER_PRICES
//
// series:
//   US_CPI_ALLITEMS_SA_M   category=CPI_ALL_ITEMS  geography=USA  is_default=true
//   US_CPI_ALLITEMS_NSA_M  category=CPI_ALL_ITEMS  geography=USA  is_default=false
//   US_CPI_ALLITEMS_YOY_M  category=CPI_ALL_ITEMS  geography=USA  is_default=false
//   JP_CPI_ALLITEMS_SA_M   category=CPI_ALL_ITEMS  geography=JPN  is_default=true
//
// Same series published by two providers = ONE series, TWO series_sources,
// optionally appearing in TWO source_groups (FRED table, OECD dataset). Its
// decomposition lives ONCE in series_hierarchy_edges, shared by both.
//
// Queries the old spine answered, now:
//   old US_CPI indicator   = series WHERE category=CPI_ALL_ITEMS AND geography=USA
//   "CPI everywhere"       = series WHERE category=CPI_ALL_ITEMS
//   "everything in Prices" = series WHERE category IN descendants(PRICES)   (recursive CTE over category_edges)
//   topic(series)          = ancestors(series.category)                     (recursive CTE) -> CONSUMER_PRICES, PRICES
//   default US CPI reading = series WHERE category=CPI_ALL_ITEMS AND geography=USA AND is_default
// =============================================================
