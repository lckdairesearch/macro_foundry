// =============================================================
// macrodb — Schema V8 (category-tree collapse + provider source groups)
// Layout: tables ordered root → leaf, all FKs written child > parent
// This gives eraser.io a consistent signal for top-down auto-layout
//
// Note: this is the canonical source of truth for the schema.
// SQLAlchemy models, Alembic migrations, Pydantic schemas, and tests must
// agree with this file. If a discrepancy is found, this is authoritative.
//
// Dependency levels:
//   L0 (roots):   geographies, categories, providers, ingestion_feeds
//   L1:           series, category_edges, provider_catalogs
//   L2:           series_hierarchy_edges, series_sources, derived_series, source_groups
//   L3:           ingestion_feed_members, derivation_inputs, source_group_members
//   L4:           ingestion_run_logs, computation_run_logs
//   L5:           ingestion_run_log_members
//   L6:           observations
//   Governance:   change_proposals, change_proposal_items
//
// V8 changes over V7 (ADR 0025):
//   - REMOVED concepts, indicators, indicator_variants, tags, concept_tags.
//   - ADDED categories (one tree; kind=topic|concept; concept node carries the
//     old concept embedding) + category_edges (strict tree, UNIQUE(child)).
//   - series GAINS category_id (nullable FK to a kind=concept node, the
//     "lowest matchable level"). (is_default was added then deferred — ADR 0027.)
//   - indicator (US_CPI) is now a DERIVED query (category_id, geography_id), not
//     a stored row; indicator-grain embedding/description retired.
//   - ADDED source_groups (typed, self-nesting provider publication unit) +
//     source_group_members (link series_sources). series_hierarchy_edges KEPT.
//   - change_proposal_items.target_type: dropped {concepts, indicators, tags,
//     indicator_variants}; added {categories, source_groups}.
//
// Settled rules carried from V7 (unchanged):
//   - No native pg ENUMs; values enforced via named CHECK constraints.
//   - No DB triggers; updated_at via ORM onupdate.
//   - The "series attaches only to a kind=concept node" rule is APP-ENFORCED
//     (Pydantic + service), not a DB constraint (ADR 0025 §3).
//
// Color legend (by conceptual layer, for eraser.io rendering):
//   orange = geography            (geographies, geography_memberships)
//   purple = catalog identity     (categories, category_edges)
//   green  = series core          (series, series_hierarchy_edges)
//   blue   = provider & publication (providers, provider_catalogs, series_sources,
//                                    source_groups, source_group_members)
//   teal   = ingestion runtime    (ingestion_feeds + members + run logs)
//   yellow = derivation           (derived_series, derivation_inputs, computation_run_logs)
//   red    = observations
//   gray   = governance           (change_proposals, change_proposal_items)
// =============================================================

// L0 — Root / Reference and Request Tables

geographies [icon: globe, color: orange] {
  id uuid pk
  code string unique              // ISO 3166-1 alpha-3 for countries (USA, GBR, FRA) ISO 3166-2 for subnational (US-CA, US-NY) WB code for cross-country aggregates (EMU, WLD, EAS) internal slug for subnational regions and custom blocs (JP-KANTO, US-MIDWEST, G7, OECD, BRICS)
  name string
  alt_name text[]                 // nullable
  type string                     // ENUM: country | subnational | subnational_region | region | bloc | world
  code_standard string            // ENUM: ISO 3166-1 | ISO 3166-2 | WB | internal
  parent_geography_id uuid                  // nullable; FK -> geographies.id Country anchor for subnational and subnational_region rows. Subnational rows point to their country as an administrative parent (US-CA > USA). Subnational-region rows point to their country as a scope anchor (JP-KANTO > JPN). Membership of subnationals into subnational regions lives in geography_memberships. Countries, cross-country regions, blocs, and world have parent = NULL; type IN ('subnational', 'subnational_region') → parent_geography_id IS NOT NULL
  notes string                    // nullable
  created_at timestamp
  updated_at timestamp
}

geography_memberships [icon: user-round, color: orange] {
  id uuid pk
  member_geography_id uuid fk   // > geographies.id  (e.g. JPN)
  group_geography_id  uuid fk   // > geographies.id  (e.g. G7)
  start_date date               // nullable
  end_date date                 // nullable
  created_at timestamp
  updated_at timestamp
}

// One tree replacing V7 concepts + tags + concept_tags (and, via series.category_id,
// the addressable role of indicators + indicator_variants).
categories [icon: folder-tree, color: purple] {
  id uuid pk
  code string unique              // canonical UPPERCASE SCREAMING_SNAKE; PRICES, CONSUMER_PRICES, CPI_ALL_ITEMS
  name string                     // human display label
  description string              // nullable
  kind string                     // ENUM: topic | concept
                                  //   topic   = browse/navigation node, NOT attachable by series (PRICES)
                                  //   concept = attachable economic idea, the old `concept` grain (CPI_ALL_ITEMS)
                                  //   Series-attach-to-concept-only is enforced APP-SIDE (Pydantic/service), not DB.
                                  //   Depth ≤ 3 is a curation convention, not a CHECK.
  embedding vector                // nullable; 1536-dim; semantic search; populated for kind=concept; HNSW cosine index ix_categories_embedding_hnsw
  embedding_model string          // nullable; model id that produced the embedding
  embedding_input_hash string     // nullable; hash of embedded input text, for staleness detection
  created_at timestamp
  updated_at timestamp
}

providers [icon: building, color: blue] {

  type string                     // ENUM: official | international_organization | vendor | internal | other

  doc_url string                  // nullable; technical documentation page
  base_url string                 // nullable; API root domain e.g. https://api.worldbank.org/v2/
  credentials_ref string           // nullable
  auth_scheme string              // nullable; ENUM: bearer_header | query_param | header_custom | basic_auth | none
  rate_limit_config jsonb         // nullable; provider-level request caps/tier metadata
  notes string                    // nullable
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

ingestion_feeds [icon: rss, color: teal] {

  request_params jsonb            // nullable; request-level parameters shared by one or more feed members
  file_path_pattern string        // nullable; e.g. /data/uploads/wb_wdi_*.csv
  response_mapping jsonb          // nullable; e.g. {"date_field": "date", "value_field": "value", "data_path": "observations"}

series [icon: activity, color: green] {
  id uuid pk
  code string unique
  name string
  alt_name text[]                 // nullable; curated aliases (provider title, informal names) used as a search aid; provider-agnostic, distinct from series_sources.external_name
  description string              // nullable but strongly encouraged
  embedding vector                // nullable; 1536-dim; semantic search; HNSW cosine index ix_series_embedding_hnsw
  embedding_model string          // nullable; model id that produced the embedding
  embedding_input_hash string     // nullable; hash of embedded input text, for staleness detection
  origin_type string              // ENUM: ingested | derived | both
  category_id uuid fk             // nullable; FK -> categories.id; the MOST-SPECIFIC (kind=concept) node this series belongs to ("lowest matchable level"). Nullable allows draft/unclassified series. Concept-only rule enforced APP-SIDE.
  // is_default was added (0019) then deferred (0021, ADR 0027): an unenforced default-reading marker, premature while concept+geography slices are single-series. Returns under its own ADR when a real multi-series slice needs it.
  geography_id uuid fk            // FK -> geographies.id
  frequency string                // ENUM: D | W | M | Q | S | A; S for semi-annual
  temporal_stock_flow string      // ENUM: stock | flow | rate | index | unknown
  unit_kind string                // ENUM: index | percent | bps | currency | count | quantity | ratio | none
  unit_scale string               // ENUM: one | thousand | million | billion | trillion
  unit_label string               // nullable; e.g. persons, jobs, barrels; mainly count | quantity
  price_basis string              // nullable; ENUM: nominal | real | ppp | other; mainly currency
  currency_code string            // nullable; ISO 4217; mainly currency; unit_kind='currency' → currency_code IS NOT NULL
  measure string                  // ENUM: level | growth | change
  measure_horizon string          // nullable; ENUM: ytd | wow | mom | qoq | yoy | custom; measure='growth' → measure_horizon IS NOT NULL
  annualized boolean              // whether the value is expressed as an annualized rate
  seasonal_adjustment string      // ENUM: SA | SAAR | NSA | unknown
  reference_kind string           // nullable; ENUM: index_base | constant_prices | ppp_base | other
  reference_year int              // nullable; e.g. 2019
  reference_label string          // nullable; e.g. "2019=100", "1982-1984=100", "2015 constant prices"
  replaced_by_series_id uuid      // nullable; FK -> series.id; set when this series is superseded
  start_date date                 // nullable
  end_date date                   // nullable; null = ongoing
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

// Parent/child relation for the categories tree. STRICT TREE via
// UNIQUE(child_category_id) = at most one parent per node. Roots have no row.
// Drop that UNIQUE to allow a DAG later (ADR 0025 §2).
category_edges [icon: git-branch, color: purple] {
  id uuid pk
  parent_category_id uuid fk      // FK -> categories.id; parent_category_id != child_category_id
  child_category_id uuid fk       // FK -> categories.id; UNIQUE(child_category_id) -> single parent = tree
  sort_order int                  // nullable; stable sibling ordering for browse
  created_at timestamp
  updated_at timestamp
}

// Provider-side PUBLICATION unit (FRED release, release table, dataset, dashboard).

// =============================================================
// Relationships
// All written as child.fk > parent.id (arrow points child → parent)
// `ON DELETE` annotations below are canonical for Phase 5 models and Phase 6 migrations.
// =============================================================

// ---- Geography hierarchy ----
geographies.parent_geography_id > geographies.id // ON DELETE RESTRICT
geographies.id < geography_memberships.member_geography_id // ON DELETE CASCADE
geographies.id < geography_memberships.group_geography_id // ON DELETE CASCADE

// ---- Category tree ----
category_edges.parent_category_id > categories.id // ON DELETE RESTRICT
category_edges.child_category_id > categories.id // ON DELETE CASCADE

// ---- Series core (anchor table + direct attributes) ----
series.category_id > categories.id // ON DELETE RESTRICT; kind=concept enforced app-side
series.geography_id > geographies.id // ON DELETE RESTRICT
series.replaced_by_series_id - series.id // ON DELETE RESTRICT
series_hierarchy_edges.parent_series_id > series.id // ON DELETE RESTRICT
series_hierarchy_edges.child_series_id > series.id // ON DELETE RESTRICT

// ---- Provider / ingestion chain ----
provider_catalogs.provider_id > providers.id // ON DELETE RESTRICT
series_sources.provider_catalog_id > provider_catalogs.id // ON DELETE RESTRICT
series_sources.series_id > series.id // ON DELETE CASCADE
ingestion_feed_members.ingestion_feed_id > ingestion_feeds.id // ON DELETE CASCADE
ingestion_feed_members.series_source_id - series_sources.id // ON DELETE CASCADE
ingestion_run_logs.ingestion_feed_id > ingestion_feeds.id // ON DELETE RESTRICT
ingestion_run_log_members.ingestion_run_log_id > ingestion_run_logs.id // ON DELETE RESTRICT
ingestion_run_log_members.ingestion_feed_member_id > ingestion_feed_members.id // ON DELETE RESTRICT

// ---- Provider source groups (publication layout) ----
source_groups.provider_catalog_id > provider_catalogs.id // ON DELETE RESTRICT
source_groups.parent_group_id > source_groups.id // ON DELETE RESTRICT
source_group_members.source_group_id > source_groups.id // ON DELETE CASCADE
source_group_members.series_source_id > series_sources.id // ON DELETE CASCADE

// ---- Derivation chain ----
derived_series.series_id - series.id // ON DELETE CASCADE
derivation_inputs.derived_series_id > derived_series.id // ON DELETE CASCADE
derivation_inputs.input_series_id > series.id // ON DELETE RESTRICT
computation_run_logs.derived_series_id > derived_series.id // ON DELETE RESTRICT

// ---- Observations (joins ingestion + derivation back to series) ----
observations.series_id > series.id // ON DELETE RESTRICT
observations.ingestion_run_log_member_id > ingestion_run_log_members.id // ON DELETE RESTRICT
observations.computation_run_log_id > computation_run_logs.id // ON DELETE RESTRICT

// ---- Governance (intentionally isolated) ----
change_proposals.superseded_by_proposal_id > change_proposals.id // ON DELETE RESTRICT
change_proposal_items.proposal_id > change_proposals.id // ON DELETE CASCADE

// =============================================================
// DERIVED (no longer stored) — how to recover the old entities
//   indicator (US_CPI)   = series WHERE category_id = <concept node> AND geography_id = <geo>
//   "CPI everywhere"      = series WHERE category_id = <CPI_ALL_ITEMS>
//   "everything in Prices"= series WHERE category_id IN descendants(<PRICES>)  -- recursive CTE over category_edges
//   topic(series)         = ancestors(series.category_id)                      -- recursive CTE over category_edges
//   (default reading: deferred — series.is_default removed in 0021, ADR 0027)
// =============================================================
