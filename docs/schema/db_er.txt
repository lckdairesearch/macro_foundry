// =============================================================
// macrodb — Schema V7 (semantic-search embeddings + governance audit vocabulary)
// Layout: tables ordered root → leaf, all FKs written child > parent
// This gives eraser.io a consistent signal for top-down auto-layout
//
// Dependency levels:
//   L0 (roots):   geographies, concepts, tags, providers, ingestion_feeds
//   L1:           series, indicators, provider_catalogs
//   L2:           concept_tags, indicator_variants, series_hierarchy_edges, series_sources, derived_series
//   L3:           ingestion_feed_members, derivation_inputs
//   L4:           ingestion_run_logs, computation_run_logs
//   L5:           ingestion_run_log_members
//   L6:           observations
//   Governance:   change_proposals, change_proposal_items
//
// Note: this is the canonical source of truth for the schema.
// SQLAlchemy models, Alembic migrations, Pydantic schemas, and tests must
// agree with this file. If a discrepancy is found, this is authoritative.
//
// V7 adds (over V6 member-level observation provenance):
//   - pgvector embeddings on concepts, indicators, series — embedding vector(1536)
//     + embedding_model + embedding_input_hash; all nullable, populated out-of-band;
//     each backed by an HNSW cosine index (ix_<table>_embedding_hnsw)
//   - change_proposals.applied_by + change_proposals.source_agent_session_id audit columns
//   - widened governance enums: target_type adds credential_ref | enum_value;
//     action adds suggest_human_apply | suggest_credential_provisioning | suggest_enum_addition;
//     validation_status adds pending_human_apply | applied_by_operator | declined_by_operator
//
// Intentionally NOT enforced (decisions to keep, not gaps):
//   - No `reference_kind IS NOT NULL → reference_year IS NOT NULL` CHECK
//   - No partial unique on indicator_variants(indicator_id) WHERE is_default=true
// =============================================================

// =======================
// L0 — Root / Reference and Request Tables
// =======================

geographies [icon: globe, color: orange] {
  id uuid pk
  code string unique              // ISO 3166-1 alpha-3 for countries (USA, GBR, FRA) ISO 3166-2 for subnational (US-CA, US-NY) WB code for cross-country aggregates (EMU, WLD, EAS) internal slug for subnational regions and custom blocs (JP-KANTO, US-MIDWEST, G7, OECD, BRICS)
  name string
  alt_name text[]                 // nullable
  type string                     // ENUM: country | subnational | subnational_region | region | bloc | world
  code_standard string            // ENUM: ISO 3166-1 | ISO 3166-2 | WB | internal
  parent_geography_id uuid                  // nullable; FK -> geographies.id Country anchor for subnational and subnational_region rows. Subnational rows point to their country as an administrative parent (US-CA > USA). Subnational-region rows point to their country as a scope anchor (JP-KANTO > JPN). Membership of subnationals into subnational regions lives in geography_memberships. Countries, cross-country regions, blocs, and world have parent = NULL; type IN ('subnational', 'subnational_region') → parent_geography_id IS NOT NULL
  notes string                    // nullable
  created_at timestamp
  updated_at timestamp
}

geography_memberships [icon: user-round, color: orange] {
  id uuid pk
  member_geography_id uuid fk   // > geographies.id  (e.g. JPN)
  group_geography_id  uuid fk   // > geographies.id  (e.g. G7)
  start_date date               // nullable
  end_date date                 // nullable
  created_at timestamp
  updated_at timestamp
}

concepts [icon: brain, color: purple] {
  id uuid pk
  code string unique              // e.g. CPI, GDP, UNEMPLOYMENT_RATE, CURRENT_ACCOUNT_BALANCE
  name string                     // curated taxonomy; added through API/admin/bootstrap flows
  description string              // nullable
  embedding vector                // nullable; 1536-dim; semantic search; HNSW cosine index ix_concepts_embedding_hnsw
  embedding_model string          // nullable; model id that produced the embedding
  embedding_input_hash string     // nullable; hash of embedded input text, for staleness detection
  created_at timestamp
  updated_at timestamp
}

tags [icon: tag, color: orange] {
  id uuid pk
  code string unique              // canonical UPPERCASE key, modeled like concepts (no code_standard)
  name string                     // human display label
  created_at timestamp
  updated_at timestamp
}

providers [icon: building, color: blue] {
  id uuid pk
  name string unique             // e.g. World Bank, International Monetary Fund
  alt_name text[]                 // nullable
  type string                     // ENUM: official | international_organization | vendor | internal | other
  homepage_url string             // nullable
  doc_url string                  // nullable; technical documentation page
  base_url string                 // nullable; API root domain e.g. https://api.worldbank.org/v2/
  credentials_ref string           // nullable
  auth_scheme string              // nullable; ENUM: bearer_header | query_param | header_custom | basic_auth | none
  rate_limit_config jsonb         // nullable; provider-level request caps/tier metadata
  notes string                    // nullable
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

ingestion_feeds [icon: rss, color: blue] {
  id uuid pk
  feed_method string              // ENUM: api | file | scrape
  endpoint_url string             // nullable
  request_params jsonb            // nullable; request-level parameters shared by one or more feed members
  file_path_pattern string        // nullable; e.g. /data/uploads/wb_wdi_*.csv
  response_mapping jsonb          // nullable; e.g. {"date_field": "date", "value_field": "value", "data_path": "observations"}
  cron_schedule string            // nullable; metadata for an external scheduler, not in-DB cron
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

series [icon: activity, color: green] {
  id uuid pk
  code string unique
  name string
  alt_name text[]                 // nullable; curated aliases (provider title, informal names) used as a search aid; provider-agnostic, distinct from series_sources.external_name
  description string              // nullable but strongly encouraged
  embedding vector                // nullable; 1536-dim; semantic search; HNSW cosine index ix_series_embedding_hnsw
  embedding_model string          // nullable; model id that produced the embedding
  embedding_input_hash string     // nullable; hash of embedded input text, for staleness detection
  origin_type string              // ENUM: ingested | derived | both
  geography_id uuid fk            // FK -> geographies.id
  frequency string                // ENUM: D | W | M | Q | S | A; S for semi-annual
  temporal_stock_flow string      // ENUM: stock | flow | rate | index | unknown
  unit_kind string                // ENUM: index | percent | bps | currency | count | quantity | ratio | none
  unit_scale string               // ENUM: one | thousand | million | billion | trillion
  unit_label string               // nullable; e.g. persons, jobs, barrels; mainly count | quantity
  price_basis string              // nullable; ENUM: nominal | real | ppp | other; mainly currency
  currency_code string            // nullable; ISO 4217; mainly currency; unit_kind='currency' → currency_code IS NOT NULL
  measure string                  // ENUM: level | growth | change
  measure_horizon string          // nullable; ENUM: ytd | wow | mom | qoq | yoy | custom; measure='growth' → measure_horizon IS NOT NULL
  annualized boolean              // whether the value is expressed as an annualized rate
  seasonal_adjustment string      // ENUM: SA | SAAR | NSA | unknown
  reference_kind string           // nullable; ENUM: index_base | constant_prices | ppp_base | other
  reference_year int              // nullable; e.g. 2019
  reference_label string          // nullable; e.g. "2019=100", "1982-1984=100", "2015 constant prices"
  replaced_by_series_id uuid      // nullable; FK -> series.id; set when this series is superseded
  start_date date                 // nullable
  end_date date                   // nullable; null = ongoing
  is_active boolean
  created_at timestamp
  updated_at timestamp
}

indicators [icon: layers, color: purple] {
  id uuid pk
  code string unique              // e.g. JP_CPI, US_CPI, JP_HOUSEHOLD_CONSUMPTION
  name string
  description string              // nullable
  embedding vector                // nullable; 1536-dim; semantic search; HNSW cosine index ix_indicators_embedding_hnsw
  embedding_model string          // nullable; model id that produced the embedding
  embedding_input_hash string     // nullable; hash of embedded input text, for staleness detection
  concept_id uuid fk              // FK -> concepts.id
  geography_id uuid fk            // FK -> geographies.id
  created_at timestamp
  updated_at timestamp
}

provider_catalogs [icon: list, color: blue] {

  provider_id uuid fk             // FK -> providers.id
  name string                     // e.g. World Development Indicators | FRED default catalog
  catalog_url string              // nullable
  doc_url string                  // nullable
  notes string                    // nullable
  is_placeholder boolean              // true when provider has no meaningful sub-catalogue e.g. FRED
  created_at timestamp
  updated_at timestamp
}

indicator_variants [icon: list, color: purple] {
  indicator_id uuid pk fk         // FK -> indicators.id
  series_id uuid pk fk           // FK -> series.id

series_hierarchy_edges [icon: git-branch, color: purple] {
  id uuid pk
  parent_series_id uuid fk        // FK -> series.id; parent_series_id != child_series_id
  child_series_id uuid fk         // FK -> series.id; unique on parent_series_id, child_series_id
  sort_order int                  // nullable; stable sibling ordering when curated
  notes string                    // nullable
  created_at timestamp
  updated_at timestamp
}

series_sources [icon: link, color: blue] {
  id uuid pk
  series_id uuid fk               // FK -> series.id
  provider_catalog_id uuid fk     // FK -> provider_catalogs.id
  external_code string            // nullable; non-unique best-effort provider locator e.g. NY.GDP.MKTP.CD | CPIAUCSL
  external_name string            // nullable; for audit and reconciliation
  ref_url string                  // nullable; human-facing source or documentation URL
  priority int                    // lower = higher priority; ordinal only; default to 1 if no other sources, if multiple sources, confirm with human
  provider_role string            // ENUM: primary_source | redistributor | harmonized | vendor_estimate | internal | other
  value_transform jsonb           // nullable; e.g. {"op": "divide", "by": 100}
  start_date date                 // nullable
  end_date date                   // nullable; null = still active
  created_at timestamp
  updated_at timestamp
}

derived_series [icon: cpu, color: yellow] {
  id uuid pk

  formula_config jsonb            // nullable
  description string              // e.g. "10Y minus 2Y Treasury yield"
  execution_policy string         // ENUM: scheduled | upstream_update | manual
  is_deterministic boolean
  requires_vintage_awareness boolean
  code_ref string                 // nullable; e.g. transformations.growth.yoy
  created_at timestamp
  updated_at timestamp
}

ingestion_run_logs [icon: clock, color: blue] {

  ingestion_feed_id uuid fk       // FK -> ingestion_feeds.id

  rows_skipped int                // nullable
  error_message string            // nullable

observations [icon: database, color: red] {
  id uuid pk

  period_start date               // first day of the period
  period_end date                 // last day of the period; period_end >= period_start

  vintage_date date               // date this vintage/release became publicly available
  ingestion_run_log_member_id uuid // nullable; FK -> ingestion_run_log_members.id
  computation_run_log_id uuid     // nullable; FK -> computation_run_logs.id
  created_at timestamp
}

change_proposals [icon: clipboard, color: purple] {
  id uuid pk
  title string
  proposal_type string            // ENUM: add_provider_series | add_derived_series | add_indicator | add_concept | code_and_db_change | schema_change | mixed
  status string                   // ENUM: proposed | approved | applied | rejected | failed | superseded
  requested_by string             // ENUM: user | agent | system
  created_by_agent string         // nullable; specific agent identifier when requested_by = agent e.g. "claude-opus-4-7", "gpt-4o"
  user_prompt string              // nullable; original request that triggered this proposal
  rationale string                // nullable
  risk_level string               // ENUM: low | medium | high
  review_notes string             // nullable
  approved_by string              // nullable
  approved_at timestamp           // nullable
  applied_at timestamp            // nullable
  applied_by string               // nullable; operator/agent that applied the proposal
  source_agent_session_id string  // nullable; originating agent session for traceability
  superseded_by_proposal_id uuid  // nullable; FK -> change_proposals.id; set when status = superseded
  created_at timestamp
  updated_at timestamp
}

change_proposal_items [icon: list-checks, color: purple] {
  id uuid pk
  proposal_id uuid fk             // FK -> change_proposals.id
  item_type string                // ENUM: db_row | code_change | migration | test | validation | documentation
  target_type string              // ENUM: concepts | series | indicators | series_sources | ingestion_feeds | file | function | test | providers | provider_catalogs | derived_series | derivation_inputs | geographies | tags | indicator_variants | geography_memberships | credential_ref | enum_value
  action string                   // ENUM: insert | update | deactivate | create_file | modify_file | run_test | validate | suggest_human_apply | suggest_credential_provisioning | suggest_enum_addition
  target_id uuid                  // nullable; existing DB row if update or deactivate; polymorphic, no FK enforcement
  target_ref string               // nullable; file path, function path, artifact path
  proposed_data jsonb             // nullable; proposed DB row or structured change summary
  diff_summary string             // nullable
  validation_status string        // ENUM: pending | passed | failed | warning | not_required | pending_human_apply | applied_by_operator | declined_by_operator
  validation_notes string         // nullable
  created_at timestamp
  updated_at timestamp
}

// =============================================================
// Relationships
// All written as child.fk > parent.id (arrow points child → parent)
// Gives eraser.io a consistent signal: parents render above children
// `ON DELETE` annotations below are canonical for Phase 5 models and Phase 6 migrations.
// =============================================================

// ---- Geography hierarchy ----
geographies.parent_geography_id > geographies.id // ON DELETE RESTRICT
geographies.id < geography_memberships.member_geography_id // ON DELETE CASCADE
geographies.id < geography_memberships.group_geography_id // ON DELETE CASCADE

// ---- Series core (anchor table + direct attributes) ----
series.geography_id > geographies.id // ON DELETE RESTRICT
series.replaced_by_series_id - series.id // ON DELETE RESTRICT

// ---- Concept / indicator cluster ----
indicators.concept_id > concepts.id // ON DELETE RESTRICT
concept_tags.concept_id > concepts.id // ON DELETE CASCADE
concept_tags.tag_id > tags.id // ON DELETE CASCADE
indicators.geography_id > geographies.id // ON DELETE RESTRICT
indicator_variants.indicator_id > indicators.id // ON DELETE CASCADE
indicator_variants.series_id - series.id // ON DELETE CASCADE
series_hierarchy_edges.parent_series_id > series.id // ON DELETE RESTRICT
series_hierarchy_edges.child_series_id > series.id // ON DELETE RESTRICT

// ---- Provider / ingestion chain ----
provider_catalogs.provider_id > providers.id // ON DELETE RESTRICT
series_sources.provider_catalog_id > provider_catalogs.id // ON DELETE RESTRICT
series_sources.series_id > series.id // ON DELETE CASCADE
ingestion_feed_members.ingestion_feed_id > ingestion_feeds.id // ON DELETE CASCADE
ingestion_feed_members.series_source_id - series_sources.id // ON DELETE CASCADE
ingestion_run_logs.ingestion_feed_id > ingestion_feeds.id // ON DELETE RESTRICT
ingestion_run_log_members.ingestion_run_log_id > ingestion_run_logs.id // ON DELETE RESTRICT
ingestion_run_log_members.ingestion_feed_member_id > ingestion_feed_members.id // ON DELETE RESTRICT

// ---- Derivation chain ----
derived_series.series_id - series.id // ON DELETE CASCADE
derivation_inputs.derived_series_id > derived_series.id // ON DELETE CASCADE
derivation_inputs.input_series_id > series.id // ON DELETE RESTRICT
computation_run_logs.derived_series_id > derived_series.id // ON DELETE RESTRICT

// ---- Observations (joins ingestion + derivation back to series) ----
observations.series_id > series.id // ON DELETE RESTRICT
observations.ingestion_run_log_member_id > ingestion_run_log_members.id // ON DELETE RESTRICT
observations.computation_run_log_id > computation_run_logs.id // ON DELETE RESTRICT

// ---- Governance (intentionally isolated) ----
change_proposals.superseded_by_proposal_id > change_proposals.id // ON DELETE RESTRICT
change_proposal_items.proposal_id > change_proposals.id // ON DELETE CASCADE
